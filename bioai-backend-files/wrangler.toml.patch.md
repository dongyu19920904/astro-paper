# ============================================================
# wrangler.toml 修改说明
# ============================================================
# 需要在 CloudFlare-BioAI-Daily/wrangler.toml 中添加以下配置

# ============================================================
# 修改 1: 添加新的 Cron 触发器
# ============================================================
# 找到 [triggers] 部分，修改 crons 配置

# 原配置:
# [triggers]
# crons = ["0 23 * * *"]

# 改为:
# [triggers]
# crons = ["0 23 * * *", "0 1 * * *"]
#
# 说明:
# - 0 23 * * * = UTC 23:00 (北京 07:00 次日) - BioAI 日报生成
# - 0 1 * * *  = UTC 01:00 (北京 09:00) - 博客生成任务（确保日报已生成）


# ============================================================
# 修改 2: 在 [vars] 部分添加博客仓库配置
# ============================================================
# 在 [vars] 部分末尾添加:

# ========================
# 博客自动生成配置
# ========================
# 博客仓库名（与 BioAI 前端分开）
BLOG_REPO_NAME = "astro-paper"
BLOG_REPO_BRANCH = "main"
